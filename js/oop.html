一、构建javascript类

function Bird(){
    var so="so";
    console.log(this);

Bird.prototype.fly=function(){

总结：使用new来创建Bird时，会新建一个object，并让this指向这个object，同时会给新生成的object对象的_proto_属性进行赋值，值为Bird的prototype属性的值，最后返回this

二、实现继承

  • 利用原型
  • 利用call
  • 组合方式
  • 寄生方式

1.利用原型的方式

function A(){
    var temp=1;
    this.apple="big";
    this.banana="small";
    this.eat=function(){
        console.log(eat);
    }
}

A.prototype.status="ok";
A.prototype.run=function(){
    console.log("减肥");
}

function B(){
    this.name="w";
}

B.prototype=new A();

var b = new B();
console.log(b.apple)
b.run();

通过原型链的方式实现伪继承，实际上是通过new A()生成一个object,然后将其当作B生成对象的_prop_属性指向的对象，
导致的问题是_proto_对象中的变量为所有实例所共享（对继承而来的属性赋值实际上是在子类中创建了一个新的属性,而对于list的push操作所引起的变化，则为所有实例所共享）

2.利用call

function C(){
    this.name="c";
    A.call(this);
}

var c = new C();
console.log(c.apple+"c");

使用call的方式，传递this对象，这样可以保证属性确实是在一个对象内,缺点是使用方法调用不会将A的portotype属性的值赋值给this的_proto_，也就是说丢失了原型链

3.使用组合的方式

function D(){
    this.name="d";
    A.call(this);
}

D.prototype=new A();

var d = new D();
console.log(d.apple)
d.run();

组合方式综合使用了call和原型链，利用call避免了变量共享，使用原型链避免了原型的丢失，缺点是，构建子类对象时，父类对象中的属性被创建了两次，一次是构建原型时，另一次是构造子类对象时。该方法能够保证继承功能的完整性，但是浪费了内存空间。函数中的this对象永远是在编写程序时直接调用该函数的对象，内部处理时会将该对象传入到this中)

4.使用寄生的方式

完美的方式，寄生组合式继承，对组合继承的改进，不必为了指定子类型的原型而调用超类型的构造函数

知识点:

  1. 因构造函数中的prototype属性而创建的原型对象，其constructor(该对象的构建者，也就是说该对象由谁创建的)的值指向构造函数对象,
  2. js中提供了Object.create的函数进行对象内存的复制

下列的函数可以直接将父类对象的原型对象直接绑定到子类对象上，处理完成之后，复制了父类的原型对象给子类，之后二者再无任何关联，实际上就是模拟了原型的构建过程

function inheritPrototyoe(subType,superType){
    var prototype = Object.create(superType.prototype);//原型对象的内存复制
    //模拟原型的构建过程
    prototype.constructor=subType;
    subType.prototype=prototype;
}

function superType(name){
    this.name=name;
    this.colors=["red","blue","green"]
}

superType.prototype.sayName=function(){
    console.log(this.name);
}

function subType(name,age){
    superType.call(this,name);
    this.age=age;
}

inheritPrototyoe(subType,superType);

subType.prototype.sayAge=function(){
    console.log(this.age);
}

var instance = new subType("Bob",18);
instance.sayName();
instance.sayAge();

总结：了解javascript的内部机制后，实现javascript继承，只有组合方式和寄生组合方式才具备可用性
